#!/data/data/com.termux/files/usr/bin/env bash

shopt -s nullglob; shopt -s globstar; set +o noclobber; IFS='\' format='audio/mp3'; format2='audio'; file="${1:+${1}}"; file2="${1:-~/downloads/*.mp3}"; files="$(eval ls -d "${file:-${file2}}")"; n="${#files[@]}"; pick="${files[RANDOM % n]}"; result="$(printf "${0:+${pick}}" | shuf -n 1)"; tmp="/sdcard/Download/"$(basename "${result}")""; \cd "/sdcard/Download" &> /dev/null; \cp -rf "${result}" "${tmp}" &> /dev/null; echo "Siap mendengarkan musik? (y/N)"; read -s -n 1 answer; echo; case "$answer" in [Yy]) echo "warning: Centang selalu aplikasi pemutar favorit anda untuk dapat melewati popup setiap kali menjalankan perintah ini."; xdg-open --content-type ${format} "${tmp}"; sleep 1; echo; cd - &> /dev/null;; *) echo -e "O-K tidak apa-apa, mungkin lain kali."; rm -f "${tmp}"; cd - &> /dev/null;; esac; #
